-- Input para definir a velocidade (posicionado primeiro)
local Input = Tabs.Misc:AddInput("SpeedInput", {
    Title = "Set Speed",
    Default = "",
    Placeholder = "Enter Speed",
    Numeric = true, -- Apenas n√∫meros
    Finished = true, -- Chama o callback quando pressionar enter
    Callback = function(Value)
        local speed = tonumber(Value)
        if speed and speed > 0 then
            customSpeed = speed -- Atualiza a velocidade personalizada
            print("Custom speed set to:", customSpeed)
            -- Aplica a velocidade imediatamente se o Toggle estiver ativo
            if Options.MyToggle.Value then
                humanoid.WalkSpeed = customSpeed
            end
        else
            print("Invalid speed value.")
        end
    end
})

        humanoid.WalkSpeed = normalSpeed -- Retorna √† velocidade padr√£o

    end
end)
